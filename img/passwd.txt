Sub UnprotectAllSheets()
' Kode ini akan mencoba semua kemungkinan kombinasi kata sandi sederhana (0 sampai 9)
' untuk membuka proteksi sheet.

Dim i As Integer
Dim j As Integer
Dim k As Integer
Dim l As Integer
Dim m As Integer
Dim n As Integer
Dim i1 As Integer, i2 As Integer, i3 As Integer, i4 As Integer, i5 As Integer, i6 As Integer

On Error Resume Next ' Melanjutkan ke baris berikutnya meskipun terjadi error (password salah)

For i = 65 To 66: For j = 65 To 66: For k = 65 To 66: For l = 65 To 66: For m = 65 To 66: For n = 65 To 66
    For i1 = 65 To 66: For i2 = 65 To 66: For i3 = 65 To 66: For i4 = 65 To 66: For i5 = 65 To 66: For i6 = 65 To 66
        ' Membangun string sandi
        ActiveSheet.Unprotect Chr(i) & Chr(j) & Chr(k) & Chr(l) & Chr(m) & Chr(n) & Chr(i1) & Chr(i2) & Chr(i3) & Chr(i4) & Chr(i5) & Chr(i6)
        
        ' Cek apakah sheet sudah tidak terproteksi
        If ActiveSheet.ProtectContents = False Then
            ' Jika sudah tidak terproteksi, tampilkan notifikasi dan keluar dari loop
            MsgBox "Sheet '" & ActiveSheet.Name & "' berhasil di-unprotect!", vbInformation
            Exit Sub ' Keluar dari sub setelah berhasil
        End If
    Next: Next: Next: Next: Next: Next
Next: Next: Next: Next: Next: Next

End Sub
